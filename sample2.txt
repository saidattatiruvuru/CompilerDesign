function int fibo(int n) {
	int first = 0, second = 1;
	int i = 0;
	while(i < n) {
		int third = first + second;
		print("\n", third);
		first = second;
		second = third;
	}
	return second;
}

int n = input(int);
int ans = fibo(n);
print(n, "th fibonacci:", ans);
float f = input(float);
float result = 0;
if(f < 0 ) {
	result = 2 * (f <= -1 && f >= -3);
	if(result == 2) {
		print("\nYay!");
	}
}
else {
	result = -2 * (f || 1);
	if(result == -2) {
		print("\nHurray");
	}
}
